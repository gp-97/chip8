\item {\bfseries{4K Memory}}
\item {\bfseries{16-\/bit Index Register}}
\item {\bfseries{16-\/bit Program Counter}}
\item {\bfseries{16-\/level Stack}}
\item {\bfseries{8-\/bit Stack Pointer}}
\item {\bfseries{8-\/bit Delay Timer}}
\item {\bfseries{8-\/bit Sound Timer}}

\end{DoxyItemize}
